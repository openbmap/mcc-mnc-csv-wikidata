PREFIX wikibase: <http://wikiba.se/ontology#>
PREFIX entity: <http://www.wikidata.org/entity/>
PREFIX wdt:      <http://www.wikidata.org/prop/direct/>
PREFIX rdfs:   <http://www.w3.org/2000/01/rdf-schema#>
PREFIX q:      <http://www.wikidata.org/prop/qualifier/>
PREFIX p:      <http://www.wikidata.org/prop/>
PREFIX ps:     <http://www.wikidata.org/prop/statement/>

SELECT ?country ?countryLabel ?mcc ?dial ?iso WHERE {
  ?country wdt:P31 entity:Q6256 .           # find instances of company
  ?country p:P2258 ?x .                       # with a P2259 (mobile network code) statement
  ?x ps:P2258 ?mcc .                           # ... that has the value ?mnc
  OPTIONAL {?country wdt:P474 ?dial} .
  OPTIONAL {?country wdt:P297 ?iso} .
  
  SERVICE wikibase:label {
    bd:serviceParam wikibase:language "en" .
  }
}
ORDER BY(?countryLabel)
